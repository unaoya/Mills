\chapter{Mills' constant}

\begin{definition}
    \label{def:mills}
    \lean{Mills}
    A positive real number $x$ is Mills if $1 < x$ and
    for all positive integers $n$, the number $\lfloor x^{3^n} \rfloor$ is prime.
\end{definition}

\begin{proposition}
    \label{prop:mills}
    \lean{exists_Mills}
    \uses{def:mills}
    There exists a Mills number.
\end{proposition}
